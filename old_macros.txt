Function FileExists(FilePath As String) As Boolean
    Dim fso As Object
    Set fso = CreateObject("Scripting.FileSystemObject")
    FileExists = fso.FileExists(FilePath)
    Set fso = Nothing
End Function

Function GetFileName(FullPath As String) As String
    Dim separator As String
    separator = Application.PathSeparator
    GetFileName = Right(FullPath, Len(FullPath) - InStrRev(FullPath, separator))
End Function

Sub Makro2()
'
' Makro am 23.05.2008 von E. Lüpfert aufgezeichnet
'

' Kopieren von Stammtabelle in Datasheet

Dim BlattName As String
Dim VorlName As String
Dim NeuAltTyp As VbMsgBoxResult
Dim DateiName As Variant
Dim NeuName As String
Dim wb As Workbook
Dim fileExtension As String

' Berechnung automatisch.
    With Application
        .Calculation = xlAutomatic
    End With

'Datasheet-Name
    ActiveCell.Rows("1:1").EntireRow.Select
    ActiveCell.Range("D1").Select
    BlattName = ActiveCell.Value

' Vorlagen-Name
    ActiveCell.Rows("1:1").EntireRow.Select
    ActiveCell.Range("X1").Select
    VorlName = ActiveCell.Value

' vorhandene Datei auswählen?
    NeuAltTyp = MsgBox("Input into existing File?", vbYesNoCancel)
    Select Case NeuAltTyp
        Case vbYes
' Name der Arbeitsmappe
            DateiName = Application.GetOpenFilename("Spectrometer-Data (*.xls; *.xlsm; *.xlsx), *.xls;*.xlsm;*.xlsx", 2, _
                        "Choose DataFile for the new Sheet", , False)
            If DateiName = False Then Exit Sub
            NeuName = GetFileName(CStr(DateiName))
            On Error Resume Next
            Set wb = Workbooks(NeuName)
            If wb Is Nothing Then
                Set wb = Workbooks.Open(Filename:=DateiName, UpdateLinks:=0)
            End If
            On Error GoTo 0
            
            If Not wb Is Nothing Then
                wb.Activate
                ThisWorkbook.Activate
                Sheets(VorlName).Copy After:=wb.Sheets(wb.Sheets.Count)
            End If
        Case vbNo
'Neue Datei
            Sheets(VorlName).Copy
            NeuName = ActiveWorkbook.Name
            
        Case vbCancel
            Exit Sub
        End Select

' Datasheet benennen
    Application.ScreenUpdating = False
    
    Sheets(VorlName).Name = BlattName
        

' Daten kopieren
    ThisWorkbook.Activate
    
    ActiveCell.Rows("1:1").EntireRow.Select
    ActiveCell.Range("E1:W1").Select
    Selection.Copy

' Daten einfügen in Mappe
    Windows(NeuName).Activate
    Sheets(BlattName).Select
'    Application.CutCopyMode = False
    Range("C5").Select
    Selection.PasteSpecial Paste:=xlPasteValues, Operation:=xlNone, SkipBlanks _
        :=False, Transpose:=True
    If IsEmpty(Range("C22")) = False Then
        Range("C22").Cut Range("C24")
    End If
' move durations/cycle to right places
    If Range("C21").Value Like "*c" Then
        Range("C21").Cut Range("D21")
        Range("D21").Replace "c", "", xlPart
        Range("C21").Value = "-"
        Range("C22").Value = "-"
    End If
' divide sample dimension in two get rid of x (10x10)
If InStr(Range("C18"), "x") = 0 Then
Else
    Dim CE As String
    Dim POS As Integer
    Dim Xdim As String
    Dim Ydim As String
    
    CE = Range("C18")
    POS = InStr(1, CE, "x")
    Xdim = Left(CE, POS - 1)
    Ydim = Right(CE, Len(CE) - POS)
    Range("C18") = Xdim
    Range("C18").NumberFormat = "0"
    Range("C18").HorizontalAlignment = xlRight
    Range("D18") = Ydim
    Range("D18").NumberFormat = "0"
    Range("D18").HorizontalAlignment = xlLeft
End If

'Blattschutz
'    ActiveSheet.Protect DrawingObjects:=True, Contents:=True, Scenarios:=True, _
'        AllowFormattingCells:=True, AllowFormattingColumns:=True, _
'        AllowFormattingRows:=True, AllowFiltering:=True, AllowUsingPivotTables:=True
    
' Speichern
'    fName = Application.GetSaveAsFilename
'Loop Until fName <> False
'NewBook.SaveAs Filename:=fName
        
        DateiName = Application.GetSaveAsFilename(NeuName, "Spektren_Datei (*.xlsm), *.xlsm", 2, _
                    "Save Spectra As (new) DataFile Name")
        If DateiName = False Then Exit Sub
        
    fileExtension = LCase(Right(CStr(DateiName), 5))
    
    If InStr(fileExtension, ".xlsm") > 0 Then
        ActiveWorkbook.SaveAs Filename:=DateiName, FileFormat:=52, _
            Password:="", WriteResPassword:="", ReadOnlyRecommended:=False, _
            CreateBackup:=False
    ElseIf InStr(fileExtension, ".xlsx") > 0 Then
        ActiveWorkbook.SaveAs Filename:=DateiName, FileFormat:=51, _
            Password:="", WriteResPassword:="", ReadOnlyRecommended:=False, _
            CreateBackup:=False
    Else
        ActiveWorkbook.SaveAs Filename:=DateiName, FileFormat:=56, _
            Password:="", WriteResPassword:="", ReadOnlyRecommended:=False, _
            CreateBackup:=False
    End If

End Sub
